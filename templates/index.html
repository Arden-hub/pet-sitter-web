let currentSessionId = "-"; function toggleChat() { const win = document.getElementById('chat-window'); win.style.display = (win.style.display === 'flex') ? 'none' : 'flex'; } function sendMessage() { const input = document.getElementById('chat-input'); const msgBox = document.getElementById('chat-messages'); const userText = input.value.trim(); if (!userText) return; // 顯示使用者訊息 msgBox.innerHTML += `
${userText}
`; input.value = ''; msgBox.scrollTop = msgBox.scrollHeight; // 顯示載入中 const loadingId = 'loading-' + Date.now(); msgBox.innerHTML += `
Kiki 正在思考中...🐾
`; msgBox.scrollTop = msgBox.scrollHeight; // 呼叫後端 API fetch('/api/kiki_chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ message: userText }) }) .then(res => res.json()) .then(data => { const loadingEl = document.getElementById(loadingId); if (loadingEl) loadingEl.remove(); msgBox.innerHTML += `
${data.reply}
`; msgBox.scrollTop = msgBox.scrollHeight; }) .catch(err => { const loadingEl = document.getElementById(loadingId); if (loadingEl) loadingEl.innerHTML = "喵嗚... 連線失敗，請檢查後端服務。"; }); }
